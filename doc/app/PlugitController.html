def logged_in_user
  if @logged_in_user
    @logged_in_user
  else
    users = User.where(ebu_id: logged_in_user_id).limit(1)
    @logged_in_user = (users.any? ? users.first : nil)
  end
end

def require_admin

def require_write_access
